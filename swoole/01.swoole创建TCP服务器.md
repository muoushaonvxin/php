
```php
<?php
// 创建Server对象
$serv = new Swoole\Server("127.0.0.1", 9501);

// 监听连接进入事件
$serv->on('connect', function ($serv, $fd){
    echo "Client:Connect.\n";
});

// 监听数据接收事件
$serv->on('receive', function ($serv, $fd, $reactor_id, $data) {
    $serv->send($fd, 'Swoole: '.$data);
    $serv->close($fd);
});

// 监听连接关闭事件
$serv->on('close', function ($serv, $fd) {
    echo "Client: Close.\n";
});

$serv->start();
```
