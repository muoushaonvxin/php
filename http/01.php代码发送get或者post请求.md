    $options = Array(
        'http' => Array(
            'method' => 'GET',
            'header' => 'Content-type:application/x-www-form-urlencoded',
            'content' => $getdata,
            'timeout' => 15 * 60
        )
    );

    $content = stream_context_create($options);

}
